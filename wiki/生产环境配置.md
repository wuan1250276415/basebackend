## JVM 调优建议

### 基础配置

```bash
# 启动参数
JAVA_OPTS="-server \
  -Xms2g -Xmx2g \
  -XX:+UseG1GC \
  -XX:MaxGCPauseMillis=200 \
  -XX:+HeapDumpOnOutOfMemoryError \
  -XX:HeapDumpPath=/logs/heapdump.hprof \
  -XX:+PrintGCDetails \
  -Xlog:gc*:file=/logs/gc.log:time,uptime:filecount=5,filesize=20M"
```

### 各服务内存推荐

| 服务 | 最小堆 | 最大堆 | 说明 |
|------|--------|--------|------|
| Gateway | 512m | 1g | 网关主要消耗连接资源 |
| User API | 1g | 2g | 常规业务服务 |
| System API | 1g | 2g | 常规业务服务 |
| Notification | 512m | 1g | SSE 连接占内存 |
| File Service | 512m | 1g | 文件上传时临时内存消耗 |
| Scheduler | 1g | 2g | 工作流引擎内存消耗 |

### G1 GC 调优

```bash
# G1 GC 推荐配置
-XX:+UseG1GC
-XX:MaxGCPauseMillis=200          # 目标暂停时间
-XX:InitiatingHeapOccupancyPercent=45  # 触发混合回收的堆占用比
-XX:G1ReservePercent=10           # 预留空间防止 to-space 溢出
-XX:ParallelGCThreads=8           # 并行 GC 线程数
-XX:ConcGCThreads=2               # 并发 GC 线程数
```

## 数据库连接池配置

### 生产环境推荐

```yaml
spring:
  datasource:
    druid:
      initial-size: 10              # 初始连接数
      min-idle: 10                  # 最小空闲连接
      max-active: 50                # 最大活跃连接
      max-wait: 60000               # 获取连接超时（ms）
      time-between-eviction-runs-millis: 60000
      min-evictable-idle-time-millis: 300000
      validation-query: SELECT 1
      test-while-idle: true
      test-on-borrow: false
      test-on-return: false
      pool-prepared-statements: true
      max-pool-prepared-statement-per-connection-size: 20
      # 监控统计
      filter:
        stat:
          enabled: true
          log-slow-sql: true
          slow-sql-millis: 1000
        wall:
          enabled: true
          config:
            multi-statement-allow: false
            drop-table-allow: false
```

### 连接数计算

```
最大连接数 = (核心数 * 2) + 有效磁盘数

示例：4 核 CPU + 1 块 SSD
最大连接数 = (4 * 2) + 1 = 9（单服务）

多服务部署时：
总连接数 = 服务数 * 每服务连接数
确保不超过 MySQL max_connections（默认 151）
```

## Redis 配置优化

### 生产环境推荐

```yaml
spring:
  data:
    redis:
      host: ${REDIS_HOST:localhost}
      port: ${REDIS_PORT:6379}
      password: ${REDIS_PASSWORD}
      database: 0
      timeout: 3000
      lettuce:
        pool:
          max-active: 32           # 最大活跃连接
          max-idle: 16             # 最大空闲连接
          min-idle: 8              # 最小空闲连接
          max-wait: 3000           # 获取连接超时
        shutdown-timeout: 100ms
```

### Redis 服务端优化

```bash
# redis.conf 关键配置
maxmemory 4gb                     # 最大内存
maxmemory-policy allkeys-lru      # 内存淘汰策略
save ""                           # 关闭 RDB（如果用 AOF）
appendonly yes                    # 启用 AOF
appendfsync everysec              # AOF 刷盘策略
tcp-keepalive 300                 # TCP 心跳
timeout 300                       # 空闲连接超时
```

## Nginx 反向代理配置

```nginx
upstream gateway {
    server 127.0.0.1:8180;
    keepalive 32;
}

server {
    listen 443 ssl http2;
    server_name api.example.com;

    ssl_certificate     /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;
    ssl_protocols       TLSv1.2 TLSv1.3;

    # 安全头
    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # API 代理
    location /api/ {
        proxy_pass http://gateway;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # 超时配置
        proxy_connect_timeout 10s;
        proxy_read_timeout 30s;
        proxy_send_timeout 30s;

        # 请求体限制
        client_max_body_size 50m;
    }

    # SSE 连接
    location /api/notifications/stream {
        proxy_pass http://gateway;
        proxy_set_header Connection '';
        proxy_http_version 1.1;
        chunked_transfer_encoding off;
        proxy_buffering off;
        proxy_cache off;
        proxy_read_timeout 86400s;  # 24 小时
    }

    # 前端静态资源
    location / {
        root /var/www/admin-web;
        try_files $uri $uri/ /index.html;

        # 静态资源缓存
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff2?)$ {
            expires 30d;
            add_header Cache-Control "public, immutable";
        }
    }
}

# HTTP 重定向到 HTTPS
server {
    listen 80;
    server_name api.example.com;
    return 301 https://$host$request_uri;
}
```

## 生产配置 Profile

```yaml
# application-prod.yml
spring:
  profiles:
    active: prod

server:
  shutdown: graceful                 # 优雅停机
  tomcat:
    max-threads: 200                 # 最大线程数
    min-spare-threads: 25            # 最小空闲线程
    accept-count: 100                # 等待队列

# 关闭调试功能
knife4j:
  enable: false                      # 关闭 API 文档

management:
  endpoints:
    web:
      exposure:
        include: health,info,prometheus   # 仅暴露必要端点
  endpoint:
    health:
      show-details: when-authorized       # 认证后才显示详情

logging:
  level:
    root: WARN
    com.basebackend: INFO
    org.springframework.security: WARN
```
